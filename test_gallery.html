document.addEventListener("DOMContentLoaded", function () { const galleryElements = document.querySelectorAll('.panoramas a'); // Create an array of items for PhotoSwipe const items = Array.from(galleryElements).map((el) => ({ src: el.getAttribute('href'), w: parseInt(el.getAttribute('data-pswp-width')), h: parseInt(el.getAttribute('data-pswp-height')) })); // Initialize PhotoSwipe when an image is clicked galleryElements.forEach((el, index) => { el.addEventListener('click', function (event) { event.preventDefault(); const pswp = new PhotoSwipe({ dataSource: items, index: index, // Enable zoom and pan panToFit: true, modal: false, closeOnScroll: false }); pswp.init(); }); }); });
